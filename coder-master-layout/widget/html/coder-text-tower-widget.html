                    const cores = ["red","blue","green","yellow","orange","purple","cyan","pink","lime"];
            let estado = [...cores];
            const cubo = document.getElementById("cubo");

            function render() {
            cubo.innerHTML = "";
            estado.forEach((cor, index) => {
                const div = document.createElement("div");
                div.classList.add("cell");
                div.style.background = cor;
                div.onclick = () => mover(index);
                cubo.appendChild(div);
            });
            }

            function mover(index) {
            const linha = Math.floor(index / 3);
            const col = index % 3;
            if (Math.random() > 0.5) {
                girarLinha(linha);
            } else {
                girarColuna(col);
            }
            render();
            }

            render();

                    function atualizarRelogio() {
            const agora = new Date();
            const h = String(agora.getHours()).padStart(2, "0");
            const m = String(agora.getMinutes()).padStart(2, "0");
            const s = String(agora.getSeconds()).padStart(2, "0");

            document.getElementById("horas").textContent = h;
            document.getElementById("minutos").textContent = m;
            document.getElementById("segundos").textContent = s;
            }

            atualizarRelogio();
            setInterval(atualizarRelogio, 1000);
